- **空行**：仅包含空白字符的行
- **总行数**：代码行 + 注释行 + 空行

## 统计工具命令

### Dart代码统计
```powershell
Get-ChildItem -Path "lib" -Recurse -Filter "*.dart" | ForEach-Object {
    $content = Get-Content $_.FullName -Encoding UTF8 -ErrorAction SilentlyContinue
    if ($content) {
        $lines = $content.Length
        $codeLines = $content | Where-Object { $_ -match '\S' -and $_ -notmatch '^\s*//' -and $_ -notmatch '^\s*/\*' -and $_ -notmatch '^\s*\*/' }
        $commentLines = $content | Where-Object { $_ -match '^\s*//' -or $_ -match '^\s*/\*' -or $_ -match '^\s*\*/' }
        $emptyLines = $content | Where-Object { $_ -notmatch '\S' }
        [PSCustomObject]@{
            File = $_.FullName
            Lines = $lines
            CodeLines = $codeLines.Count
            CommentLines = $commentLines.Count
            EmptyLines = $emptyLines.Count
        }
    }
}
```

### Python代码统计
```powershell
Get-ChildItem -Path "ml_training" -Recurse -Filter "*.py" | ForEach-Object {
    $content = Get-Content $_.FullName -Encoding UTF8 -ErrorAction SilentlyContinue
    if ($content) {
        $lines = $content.Length
        $codeLines = $content | Where-Object { $_ -match '\S' -and $_ -notmatch '^\s*#' }
        $commentLines = $content | Where-Object { $_ -match '^\s*#' }
        $emptyLines = $content | Where-Object { $_ -notmatch '\S' }
        [PSCustomObject]@{
            File = $_.FullName
            Lines = $lines
            CodeLines = $codeLines.Count
            CommentLines = $commentLines.Count
            EmptyLines = $emptyLines.Count
        }
    }
}
```
